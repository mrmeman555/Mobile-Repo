    return wave_speed_km_s / freq_hz * 1000  # convert to km

# Add sites to map with frequency-based visualization
for site in sites:
    # Color based on frequency (lower freq = red, higher = blue)
    color = f"hsl({240 - site['freq']*6}, 100%, 50%)"

        popup=f"{site['name']}: {site['freq']} Hz",
        icon=folium.Icon(color='red')
    ).add_to(world_map)
    
    # Add wavelength circle
    folium.Circle(

        radius=freq_to_wavelength(site['freq']) * 1000,  # convert to meters
        color=color,
        fill=False,
        popup=f"Wavelength at {site['freq']} Hz"
