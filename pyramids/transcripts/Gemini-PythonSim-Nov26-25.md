def simulate_serapeum_acoustics():

    # Host Medium: Limestone (The Tunnel/Bedrock)
    rho_host = 2600.0       # Density of limestone (kg/m^3)
    v_host = 2500.0         # Speed of sound in limestone (m/s)
    Bulk_host = rho_host * v_host**2 # Bulk modulus (Pa)

    # Resonator: Granite Box
    f_resonance = 20.0      # Natural frequency derived from user prompt (Hz)
    w_res = 2 * np.pi * f_resonance
    
    # Filling Fraction (F)
    # Estimate: Box volume / Unit Cell volume
    # Box ~ 3m x 2m x 2m. Tunnel segment ~ 5m x 5m x 8m. 
    # This is an estimate, but F affects the WIDTH of the gap, not the center frequency.
    F = 0.25 

    # Damping (gamma) - explicit damping to avoid singularities in plot
    gamma = 0.05 * w_res 

    # Frequency Range to Simulate
    freqs = np.linspace(1, 60, 1000) # 1 Hz to 60 Hz
    omegas = 2 * np.pi * freqs

    # --- 1. Effective Mass Density Calculation ---
    # Formula based on Liu et al. (2000) for locally resonant material
    # rho_eff = rho_host * (1 + (F * w_res^2) / (w_res^2 - w^2 - i*gamma*w))
    
    # We use complex numbers to handle phase/damping
    rho_eff = rho_host * (1 + (F * w_res**2) / (w_res**2 - omegas**2 - 1j * gamma * omegas))

    # --- 2. Transmission Calculation (1D Slab Model) ---
    # Simulating transmission through a finite length of this metamaterial (e.g., 50m tunnel)
    L_tunnel = 50.0 # meters
    
    # Wavenumber k = w * sqrt(rho_eff / Bulk_host)
    # Note: Bulk modulus is assumed relatively constant for the host matrix
    k_complex = omegas * np.sqrt(rho_eff / Bulk_host)
    
    # Transmission Coefficient T = exp(-Im(k) * L) * Phase_Factor

    # T_power = exp(-2 * Imaginary(k) * L) 
    
    transmission = np.exp(-2 * np.imag(k_complex) * L_tunnel)
